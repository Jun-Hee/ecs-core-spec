Java — STS-Adapter 구조 (Netty 기반)
@Component
public class StsSocketAdapter extends ChannelInboundHandlerAdapter {
    private volatile Channel channel;

    // ECS → STS: 송신
    public void send(String msgNo, String cmdId, String eqpNo, String eqpTp,
                     Map<String, Object> payload) {
        SeohoFrame frame = SeohoFrame.builder()
            .msgNo(msgNo).msgId(UUID.randomUUID().toString())
            .cmdId(cmdId).eqpNo(eqpNo).eqpTp(eqpTp)
            .payload(objectMapper.writeValueAsString(payload)).build();
        channel.writeAndFlush(frame.encode()); // Delimiter/Length-prefix: 스펙 수신 후 확정
    }

    // STS → ECS: 수신 및 라우팅
    @Override
    public void channelRead(ChannelHandlerContext ctx, Object msg) {
        SeohoFrame frame = (SeohoFrame) msg;
        auditLogger.log("STS_RX", frame.getRaw());  // 감사 로그 (ecs.audit.log)
        switch (frame.getMsgNo()) {
            case "S01-ARPT"  -> stateManager.updateStsState(normalize(frame));
            case "J02-ARPT", "J04-ARPT", "J05-ARPT" -> orchestrator.onArpt(frame);
            case "QC-JR001", "QC-JR002", "QC-JR003",
                 "QC-JR004", "QC-JR005"             -> orchestrator.onQcJr(frame);
            case "C02-AODR"                          -> orchestrator.onHandshake(frame);
            default -> log.warn("Unknown STS msgNo: {}", frame.getMsgNo());
        }
    }

    // 재연결
    @Override
    public void channelInactive(ChannelHandlerContext ctx) {
        log.warn("STS Socket disconnected. Scheduling reconnect...");
        ctx.executor().schedule(this::reconnect, 3000, TimeUnit.MILLISECONDS);
    }

    @Override
    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
        log.error("STS Socket error", cause);
        alarmService.critical("ANOM-005", "STS_ADAPTER", "Socket error: " + cause.getMessage());
        ctx.close();
    }
}

ETW/ETA 계산 상세

ETW — Estimated Time to Work
ETW = NOW + remainingTime + ETW_BUFFER_SEC(30s)
remainingTime = avgCycleTimeSec × (1 - currentPhase/totalPhases)

STS가 현재 작업을 완료하기까지 예상 시간

ETA — Estimated Time of Arrival
occupancy = zone.agvCount / zone.maxCapacity
congestion = occupancy > 0.8 ? 1.4 : 1.2
ETA = NOW + (pathDist/agv.speed) × congestion
// BAT_LOW(<30%): ETA += 60s 패널티

AGV가 POW(작업 위치)에 도착하기까지 예상 시간

DGT 주의사항ETW/ETA는 정밀 물리 계산이 아닌 배정 우선순위 점수로 사용됩니다. Zone 점유율은 State Manager의 zone:state:{zone_id} Redis 키에서 실시간 조회합니다. Zone 구성 및 최대 AGV 수용수 테이블은 CLT/DGT가 제공 예정.

/**
 * C02-AODR 수신 시 발신 장비 + ITV_XT_NO 분석으로 3경로 분기
 * - 경로 A: STS 발신 + AGV ID → AGV Handshake (DS/LD Quay)
 * - 경로 B: ARMG 발신 + AGV ID → ARMG-AGV Handshake (DS/LD Yard)
 * - 경로 C: ARMG 발신 + 트럭번호 → XT Handshake (GI/GO)
 */
public void onHandshakeRequest(HandshakeEvent event) {
    String eqpNo   = event.getEqpNo();
    String itvXtNo = event.getItvXtNo();   // C02-AODR.ITV_XT_NO or ITV_NO

    if (isStsEquipment(eqpNo)) {
        // 경로 A — DS/LD AGV Handshake
        // STS에서 발신, itvXtNo = AGV ID
        // 후속: STS-Adapter.send(C02-TRPT) → J04-TODR(DS) or J05-TODR(LD) via Socket
        agvHandshakeHandler.handle(itvXtNo, event);

    } else if (isArmgEquipment(eqpNo) && isAgvId(itvXtNo)) {
        // 경로 B — DS/LD ARMG-AGV Handshake
        // ARMG에서 발신, itvXtNo = AGV ID
        // 후속: ARMG-Adapter.send(C02-TRPT) → J03-TODR(DS) or J02-TODR(LD) via REST
        armgAgvHandshakeHandler.handle(itvXtNo, event);

    } else if (isArmgEquipment(eqpNo) && isTruckId(itvXtNo)) {
        // 경로 C — GI/GO XT Handshake
        // ARMG에서 발신, itvXtNo = 외부 트럭 번호
        InternalTask task = taskRepository.findActiveByArmg(eqpNo);
        if (!itvXtNo.equals(task.getTruckId())) {
            // truckId 불일치 → CRITICAL 알람 + 거부
            alarmService.critical("TRUCK_MISMATCH", eqpNo,
                "Expected: " + task.getTruckId() + " Actual: " + itvXtNo);
            armgAdapter.sendTrpt(event.getCmdId(), eqpNo, itvXtNo, "N"); // clearance=N
            return;
        }
        xtHandshakeHandler.handle(itvXtNo, event);
        // 후속: ARMG-Adapter.send(C02-TRPT, clearance=Y) → J05-TODR(GO) or J06-TODR(GI) via Socket
    }
}
